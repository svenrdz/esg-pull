  options {
    VARCHAR distrib
    VARCHAR latest
    VARCHAR replica
    VARCHAR retracted
    INTEGER id PK
    VARCHAR sha
  }

  facet {
    VARCHAR name
    VARCHAR value
    INTEGER id PK
    VARCHAR sha
  }

  select {
    INTEGER id PK
    VARCHAR sha
  }

  file {
    VARCHAR file_id
    VARCHAR dataset_id
    VARCHAR master_id
    VARCHAR url
    VARCHAR version
    VARCHAR filename
    VARCHAR local_path
    VARCHAR data_node
    VARCHAR checksum
    VARCHAR checksum_type
    INTEGER size
    VARCHAR status
    INTEGER id PK
    VARCHAR sha
  }

  tag {
    VARCHAR name
    TEXT description
    INTEGER id PK
    VARCHAR sha
  }

  query {
    BOOLEAN tracked
    VARCHAR require
    INTEGER options_id FK
    INTEGER select_id FK
    INTEGER id PK
    VARCHAR sha
  }

  query  }|--|| options : "has"
  query  }|--|| select : "selects"
  tag  }o--|{ query : "describes"
  query  }|--|{ file : "represents"
  select  }|--|{ facet : "with"

  select  }|--|| select_facet : "select_id"
  facet  }|--|| select_facet : "facet_id"
